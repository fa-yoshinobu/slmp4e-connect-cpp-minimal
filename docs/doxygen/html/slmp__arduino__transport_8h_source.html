26/**
27 * @class ArduinoClientTransport
28 * @ingroup SLMP_Transport
29 * @brief Transport adapter for Arduino 'Client' objects (TCP).
30 *
31 * Wraps classes like `WiFiClient`, `EthernetClient`, or `GSMClient`.
32 * Implements reliable stream-based communication for SLMP 3E/4E frames.
33 */
34class ArduinoClientTransport : public ITransport {
35 public:
36 /**
37 * @brief Wrap an existing Arduino Client.
38 * @param client Reference to an Arduino Client object (e.g., WiFiClient).
39 */
40 explicit ArduinoClientTransport(::Client& client) : client_(client) {}
41
42 /** @brief Connect to PLC via TCP. */
43 bool connect(const char* host, uint16_t port) override {
44 return client_.connect(host, port) == 1;
45 }
46
47 /** @brief Stop the client and close connection. */
48 void close() override {
49 client_.stop();
50 }
51
52 /** @brief Check if the TCP socket is connected. */
53 bool connected() const override {
54 return client_.connected() != 0;
55 }
56
57 /** @brief Block until all data is written to the stream. */
58 bool writeAll(const uint8_t* data, size_t length) override {
59 size_t written_total = 0;
60 while (written_total < length) {
61 size_t written_now = client_.write(data + written_total, length - written_total);
62 if (written_now == 0) {
63 return false;
64 }
65 written_total += written_now;
66 }
67 return true;
68 }
69
70 /** @brief Block until exact length is read from the stream or timeout occurs. */
71 bool readExact(uint8_t* data, size_t length, uint32_t timeout_ms) override {
72 uint32_t started_at = millis();
73 size_t received = 0;
74 while (received < length) {
75 int available_now = client_.available();
76 if (available_now > 0) {
77 int read_now = client_.read(data + received, length - received);
78 if (read_now <= 0) {
79 return false;
80 }
81 received += static_cast<size_t>(read_now);
82 continue;
83 }
84
85 if (!connected()) {
86 return false;
87 }
88 if (static_cast<uint32_t>(millis() - started_at) >= timeout_ms) {
89 return false;
90 }
91 delay(1);
92 }
93 return true;
94 }
95
96 /** @brief Non-blocking write to the stream. */
97 size_t write(const uint8_t* data, size_t length) override {
98 return client_.write(data, length);
99 }
100
101 /** @brief Non-blocking read from the stream. */
102 size_t read(uint8_t* data, size_t length) override {
103 return client_.read(data, length);
104 }
105
106 /** @brief Check number of bytes available in the stream. */
107 size_t available() override {
108 return client_.available();
109 }
110
111 private:
112 ::Client& client_;
113};

117/**
118 * @class ArduinoUdpTransport
119 * @ingroup SLMP_Transport
120 * @brief Transport adapter for Arduino 'UDP' objects.
121 *
122 * Wraps classes like `WiFiUDP` or `EthernetUDP`.
123 * Manages remote endpoint (host/port) and packet framing for SLMP.
124 *
125 * @note UDP is connectionless; "connected" state in this class simply means
126 * that `begin()` has been called and remote endpoint is known.
127 */
128class ArduinoUdpTransport : public ITransport {
129 public:
130 /**
131 * @brief Wrap an existing Arduino UDP object.
132 * @param udp Reference to UDP object.
133 * @param local_port Local port to bind to (0 = use same as remote port).
134 */
135 explicit ArduinoUdpTransport(::UDP& udp, uint16_t local_port = 0)
136 : udp_(udp), local_port_(local_port) {}
137
138 /** @brief Set remote host and begin listening on local port. */
139 bool connect(const char* host, uint16_t port) override {
140 if (host == nullptr || port == 0U) {
141 connected_ = false;
142 return false;
143 }
144
145 host_ = host;
146 remote_port_ = port;
147 const uint16_t bind_port = (local_port_ == 0U) ? port : local_port_;
148 connected_ = (udp_.begin(bind_port) == 1);
149 packet_available_ = 0;
150 return connected_;
151 }
152
153 /** @brief Stop listening and clear remote endpoint. */
154 void close() override {
155 udp_.stop();
156 connected_ = false;
157 packet_available_ = 0;
158 }
159
160 /** @brief True if the transport is initialized. */
161 bool connected() const override {
162 return connected_;
163 }
164
165 /** @brief Send a single UDP packet containing the frame. */
166 bool writeAll(const uint8_t* data, size_t length) override {
167 return sendPacket(data, length);
168 }
169
170 /** @brief Block until a UDP packet of exact length is received or timeout. */
171 bool readExact(uint8_t* data, size_t length, uint32_t timeout_ms) override {
172 const uint32_t started_at = millis();
173 size_t received = 0;
174 while (received < length) {
175 if (available() > 0U) {
176 const size_t read_now = read(data + received, length - received);
177 if (read_now == 0U) {
178 return false;
179 }
180 received += read_now;
181 continue;
182 }
183
184 if (!connected_) {
185 return false;
186 }
187 if (static_cast<uint32_t>(millis() - started_at) >= timeout_ms) {
188 return false;
189 }
190 delay(1);
191 }
192 return true;
193 }
194
195 /** @brief Non-blocking packet send. Returns @p length on success. */
196 size_t write(const uint8_t* data, size_t length) override {
197 return sendPacket(data, length) ? length : 0U;
198 }
199
200 /** @brief Non-blocking read from the current received packet. */
201 size_t read(uint8_t* data, size_t length) override {
202 if (available() == 0U) {
203 return 0U;
204 }
205 const int read_now = udp_.read(data, length);
206 if (read_now <= 0) {
207 return 0U;
208 }
209 const size_t read_size = static_cast<size_t>(read_now);
210 packet_available_ = (packet_available_ > read_size) ? (packet_available_ - read_size) : 0U;
211 return read_size;
212 }
213
214 /** @brief Check if a new packet has arrived and return its size. */
215 size_t available() override {
216 if (!connected_) {
217 return 0U;
218 }
219 if (packet_available_ == 0U) {
220 const int packet_size = udp_.parsePacket();
221 if (packet_size > 0) {
222 packet_available_ = static_cast<size_t>(packet_size);
223 }
224 }
225 return packet_available_;
226 }
227
228 private:
229 bool sendPacket(const uint8_t* data, size_t length) {
230 if (!connected_ || data == nullptr || length == 0U) {
231 return false;
232 }
233 if (udp_.beginPacket(host_.c_str(), remote_port_) != 1) {
234 connected_ = false;
235 return false;
236 }
237 const size_t written = udp_.write(data, length);
238 if (written != length || udp_.endPacket() != 1) {
239 connected_ = false;
240 return false;
241 }
242 return true;
243 }
244
245 ::UDP& udp_;
246 String host_;
247 uint16_t remote_port_ = 0;
248 uint16_t local_port_ = 0;
249 size_t packet_available_ = 0;
250 bool connected_ = false;
251};
