538bool runBench(BenchMode mode, uint32_t cycle_limit) {
539 if (!connectPlc(false)) {
540 Serial.println("bench failed: plc not connected");
541 return false;
542 }
543
544 clearBenchTargets();
545 BenchSummary summary = {};
546 const uint32_t started_ms = millis();
547 uint32_t last_report_ms = started_ms;
548 uint16_t seed = 1000;
549
550 for (uint32_t cycle = 0; cycle < cycle_limit; ++cycle) {
551 const uint32_t cycle_started_ms = millis();
552 bool ok = false;
553 size_t request_count = 0;
554
555 if (mode == BenchMode::Row) {
556 uint16_t value = 0;
557 ok = plc.readOneWord(kBenchOneWordDevice, value) == slmp::Error::Ok;
558 request_count = 1;
559 } else if (mode == BenchMode::Wow) {
560 ok = plc.writeOneWord(kBenchOneWordDevice, seed++) == slmp::Error::Ok;
561 request_count = 1;
562 } else if (mode == BenchMode::Pair) {
563 const uint16_t expected = seed++;
564 uint16_t readback = 0;
565 ok = plc.writeOneWord(kBenchOneWordDevice, expected) == slmp::Error::Ok &&
566 plc.readOneWord(kBenchOneWordDevice, readback) == slmp::Error::Ok &&
567 readback == expected;
568 request_count = 2;
569 } else if (mode == BenchMode::Rw) {
570 ok = plc.readWords(
571 kBenchWordArrayDevice,
572 static_cast<uint16_t>(kBenchWordPoints),
573 bench_word_readback,
574 kBenchWordPoints) == slmp::Error::Ok;
575 request_count = 1;
576 } else if (mode == BenchMode::Ww) {
577 fillBenchWords(bench_word_values, kBenchWordPoints, seed);
578 seed = static_cast<uint16_t>(seed + kBenchWordPoints);
579 ok = plc.writeWords(kBenchWordArrayDevice, bench_word_values, kBenchWordPoints) == slmp::Error::Ok;
580 request_count = 1;
581 } else if (mode == BenchMode::Block) {
582 fillBenchWords(bench_block_values, kBenchBlockPoints, seed);
583 seed = static_cast<uint16_t>(seed + kBenchBlockPoints);
584 const slmp::DeviceBlockWrite block = {kBenchBlockWordDevice, bench_block_values, static_cast<uint16_t>(kBenchBlockPoints)};
585 const slmp::DeviceBlockRead read_block = {kBenchBlockWordDevice, static_cast<uint16_t>(kBenchBlockPoints)};
586 ok = plc.writeBlock(&block, 1, nullptr, 0) == slmp::Error::Ok &&
587 plc.readBlock(&read_block, 1, nullptr, 0, bench_block_readback, kBenchBlockPoints, nullptr, 0) == slmp::Error::Ok &&
588 wordArraysEqual(bench_block_values, bench_block_readback, static_cast<uint16_t>(kBenchBlockPoints));
589 request_count = 2;
590 }
591
592 summary.last_cycle_ms = millis() - cycle_started_ms;
593 summary.last_request_bytes = plc.lastRequestFrameLength();
594 summary.last_response_bytes = plc.lastResponseFrameLength();
595 if (summary.cycles == 0 || summary.last_cycle_ms < summary.min_cycle_ms) {
596 summary.min_cycle_ms = summary.last_cycle_ms;
597 }
598 if (summary.last_cycle_ms > summary.max_cycle_ms) {
599 summary.max_cycle_ms = summary.last_cycle_ms;
600 }
601 summary.total_cycle_ms += summary.last_cycle_ms;
602 ++summary.cycles;
603 summary.requests += static_cast<uint32_t>(request_count);
604
605 if (!ok) {
606 ++summary.fail;
607 Serial.print("bench failed: mode=");
608 Serial.print(benchModeText(mode));
609 Serial.print(" cycle=");
610 Serial.println(cycle + 1U);
611 printLastPlcError("bench");
612 printBenchProgress("bench summary:", mode, summary, started_ms);
613 clearBenchTargets();
614 return false;
615 }
616
617 if (millis() - last_report_ms >= kBenchReportIntervalMs) {
618 printBenchProgress("bench progress:", mode, summary, started_ms);
619 last_report_ms = millis();
620 }
621 }
622
623 printBenchProgress("bench summary:", mode, summary, started_ms);
624 clearBenchTargets();
625 return true;
626}

1974void loop() {
1975 while (Serial.available() > 0) {
1976 const int raw = Serial.read();
1977 if (raw < 0) {
1978 break;
1979 }
1980 const char ch = static_cast<char>(raw);
1981 if (ch == '\r' || ch == '\n') {
1982 if (ch == '\n' && serial_last_was_cr) {
1983 serial_last_was_cr = false;
1984 continue;
1985 }
1986 serial_last_was_cr = (ch == '\r');
1987 Serial.println();
1988 submitSerialLine();
1989 continue;
1990 }
1991 serial_last_was_cr = false;
1992 if (ch == '\b' || ch == 127) {
1993 if (serial_line_length > 0) {
1994 --serial_line_length;
1995 serial_line[serial_line_length] = '\0';
1996 Serial.print("\b \b");
1997 }
1998 continue;
1999 }
2000 if (!isprint(static_cast<unsigned char>(ch)) && ch != '\t') {
2001 continue;
2002 }
2003 if (serial_line_length + 1 >= sizeof(serial_line)) {
2004 Serial.println();
2005 Serial.println("command too long");
2006 serial_line_length = 0;
2007 serial_line[0] = '\0';
2008 printPrompt();
2009 continue;
2010 }
2011 Serial.print(ch);
2012 serial_line[serial_line_length++] = ch;
2013 }
