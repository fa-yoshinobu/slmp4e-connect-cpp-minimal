29/**
30 * @class ReconnectHelper
31 * @brief Automates PLC connection maintenance.
32 *
33 * Handles periodic retry logic when the connection is lost,
34 * ensuring the application remains connected without blocking the main loop.
35 */
36class ReconnectHelper {
37 public:
38 /**
39 * @brief Initialize helper with client and target.
40 * @param client Reference to an initialized SlmpClient.
41 * @param host Target IP address.
42 * @param port Target Port.
43 * @param options Reconnection configuration.
44 */
45 ReconnectHelper(
46 SlmpClient& client,
47 const char* host,
48 uint16_t port,
49 const ReconnectOptions& options = ReconnectOptions()
50 )
51 : client_(client),
52 host_(host),
53 port_(port),
54 options_(options),
55 last_attempt_ms_(0),
56 has_attempt_(false),
57 connected_edge_(false) {}
58
59 /**
60 * @brief Check and restore connection if needed.
61 * @param now_ms Current system time in milliseconds.
62 * @return true if currently connected (or just successfully reconnected).
63 */
64 bool ensureConnected(uint32_t now_ms) {
65 connected_edge_ = false;
66 if (client_.connected()) {
67 return true;
68 }
69 if (host_ == nullptr || port_ == 0U) {
70 return false;
71 }
72 if (has_attempt_) {
73 const uint32_t elapsed = now_ms - last_attempt_ms_;
74 if (elapsed < options_.retry_interval_ms) {
75 return false;
76 }
77 }
78
79 has_attempt_ = true;
80 last_attempt_ms_ = now_ms;
81 if (!client_.connect(host_, port_)) {
82 return false;
83 }
84
85 connected_edge_ = true;
86 return true;
87 }
88
89 /**
90 * @brief Returns true if a NEW connection was established since the last call.
91 *
92 * Useful for triggering one-time initialization (e.g., reading PLC type).
93 */
94 bool consumeConnectedEdge() {
95 const bool value = connected_edge_;
96 connected_edge_ = false;
97 return value;
98 }
99
100 /**
101 * @brief Manually trigger a reconnection attempt on the next update.
102 */
103 void forceReconnect(uint32_t now_ms) {
104 client_.close();
105 connected_edge_ = false;
106 has_attempt_ = true;
107 last_attempt_ms_ = now_ms - options_.retry_interval_ms;
108 }
109
110 private:
111 SlmpClient& client_;
112 const char* host_;
113 uint16_t port_;
114 ReconnectOptions options_;
115 uint32_t last_attempt_ms_;
116 bool has_attempt_;
117 bool connected_edge_;
118};
