35/**
36 * @class ReconnectHelper
37 * @ingroup SLMP_Utils
38 * @brief Automates PLC connection maintenance.
39 *
40 * Handles periodic retry logic when the connection is lost,
41 * ensuring the application remains connected without blocking the main loop.
42 *
43 * ### Example Usage
44 * @code
45 * slmp::ReconnectHelper recon(plc, "192.168.1.10", 1025);
46 *
47 * void loop() {
48 * if (recon.ensureConnected(millis())) {
49 * if (recon.consumeConnectedEdge()) {
50 * // Do one-time setup on connect
51 * }
52 * // Normal polling
53 * }
54 * }
55 * @endcode
56 */
57class ReconnectHelper {
58 public:
59 /**
60 * @brief Initialize helper with client and target endpoint.
61 * @param client Reference to an initialized @ref SlmpClient.
62 * @param host Target IP address or hostname.
63 * @param port Target Port (usually 1025 or 1280).
64 * @param options Reconnection configuration.
65 */
66 ReconnectHelper(
67 SlmpClient& client,
68 const char* host,
69 uint16_t port,
70 const ReconnectOptions& options = ReconnectOptions()
71 )
72 : client_(client),
73 host_(host),
74 port_(port),
75 options_(options),
76 last_attempt_ms_(0),
77 has_attempt_(false),
78 connected_edge_(false) {}
79
80 /**
81 * @brief Check and restore connection if needed.
82 *
83 * If already connected, returns true immediately.
84 * If disconnected, attempts to connect only if @ref ReconnectOptions::retry_interval_ms
85 * has elapsed since the last attempt.
86 *
87 * @param now_ms Current system time in milliseconds.
88 * @return true if currently connected (or just successfully reconnected).
89 */
90 bool ensureConnected(uint32_t now_ms) {
91 connected_edge_ = false;
92 if (client_.connected()) {
93 return true;
94 }
95 if (host_ == nullptr || port_ == 0U) {
96 return false;
97 }
98 if (has_attempt_) {
99 const uint32_t elapsed = now_ms - last_attempt_ms_;
100 if (elapsed < options_.retry_interval_ms) {
101 return false;
102 }
103 }
104
105 has_attempt_ = true;
106 last_attempt_ms_ = now_ms;
107 if (!client_.connect(host_, port_)) {
108 return false;
109 }
110
111 connected_edge_ = true;
112 return true;
113 }
114
115 /**
116 * @brief Detect a rising edge of the connection state.
117 *
118 * Returns true only ONCE when a new connection is established.
119 * Useful for triggering initialization logic like reading PLC type or setting up monitors.
120 *
121 * @return true if a NEW connection was established since the last call to this method.
122 */
123 bool consumeConnectedEdge() {
124 const bool value = connected_edge_;
125 connected_edge_ = false;
126 return value;
127 }
128
129 /**
130 * @brief Manually trigger a reconnection attempt on the next call to @ref ensureConnected.
131 *
132 * Closes the current connection and resets the retry timer to allow
133 * an immediate attempt if desired.
134 */
135 void forceReconnect(uint32_t now_ms) {
136 client_.close();
137 connected_edge_ = false;
138 has_attempt_ = true;
139 last_attempt_ms_ = now_ms - options_.retry_interval_ms;
140 }
141
142 private:
143 SlmpClient& client_;
144 const char* host_;
145 uint16_t port_;
146 ReconnectOptions options_;
147 uint32_t last_attempt_ms_;
148 bool has_attempt_;
149 bool connected_edge_;
150};
