30/**
31 * @enum Error
32 * @brief Error codes returned by SLMP operations.
33 */
34enum class Error : uint8_t {
35 Ok = 0, ///< Operation successful.
36 InvalidArgument, ///< One or more arguments are invalid (e.g. nullptr or range error).
37 UnsupportedDevice, ///< The requested device code is not supported by this library or target PLC.
38 BufferTooSmall, ///< Provided buffer capacity is insufficient for request or response.
39 NotConnected, ///< Transport layer is not connected.
40 TransportError, ///< Error occurred in the transport layer (e.g., TCP timeout, UDP loss).
41 ProtocolError, ///< Malformed packet or protocol violation detected.
42 PlcError, ///< PLC returned a non-zero end code (use @ref SlmpClient::lastEndCode to get details).
43 Busy, ///< An asynchronous operation is already in progress.
44};
45
46/**
47 * @enum FrameType
48 * @brief SLMP frame formats supported by the library.
49 */
50enum class FrameType : uint8_t {
51 Frame3E, ///< Standard 3E frame (most common for Q/L/iQ-R).
52 Frame4E, ///< 4E frame (extended functionality, used for iQ-R specific features).
53};

55/**
56 * @enum CompatibilityMode
57 * @brief Subcommand selection for device access.

59 * Determines whether to use modern iQ-R extended subcommands or legacy Q/L subcommands.

61enum class CompatibilityMode : uint8_t {

63 Legacy, ///< Use subcommands 0x0000/0x0001 (Legacy Q/L series).
64};

320/**
321 * @class SlmpClient
322 * @brief Main SLMP client implementation.
323 *
324 * This class provides high-level device access methods. It is designed to be:
325 * - **Memory Efficient**: No dynamic allocation. User provides buffers.
326 * - **Flexible**: Supports synchronous (blocking) and asynchronous (non-blocking) calls.
327 * - **Robust**: Validates buffer capacities and protocol state.
328 *
329 * ### Synchronous Usage Example
330 * @code
331 * slmp::SlmpClient plc(transport, tx_buf, 256, rx_buf, 256);
332 * uint16_t val;
333 * if (plc.readOneWord(slmp::dev::D(slmp::dev::dec(100)), val) == slmp::Error::Ok) {
334 * // Success
335 * }
336 * @endcode
337 *
338 * ### Asynchronous Usage Example
339 * @code
340 * if (plc.beginReadOneWord(slmp::dev::D(slmp::dev::dec(100)), val, millis()) == slmp::Error::Ok) {
341 * while(plc.isBusy()) {
342 * plc.update(millis());
343 * }
344 * }
345 * @endcode
346 */
347class SlmpClient {
348 public:
